修改主机名

```shell
[root@localhost ~]# hostname newname
```

查看系统信息

```shell
查看系统版本
[root@localhost ~]# cat /etc/redhat-release
CentOS Linux release 7.5.1804 (Core)

查看操作系统的内核版本
[root@localhost ~]# uname -r
3.10.0-862.el7.x86_64

查看CPU型号
[root@localhost ~]# cat /proc/cpuinfo
processor	: 0
vendor_id	: GenuineIntel
cpu family	: 6
model		: 158
model name	: Intel(R) Core(TM) i5-7300HQ CPU @ 2.50GHz
```

时间命令 date

```shell
[root@localhost ~]# date
Mon Mar  6 23:28:07 CST 2023
[root@localhost ~]# date +%F
2023-03-06
[root@localhost ~]# date +%T
23:28:17
[root@localhost ~]# date +%F-%T
2023-03-06-23:28:35
[root@localhost ~]# date +%Y
2023
```

设置命令别名

```shell
# alias 别名='命令'		//临时的

[root@localhost ~]# alias cs='cat /etc/sysconfig/network-scripts/ifcfg-ens33'

//删除别名
[root@localhost ~]# unalias cs


# 修改配置文件 //永久生效
[root@localhost ~]# vim /etc/bashrc
末尾添加一行
alias cs='cat /etc/sysconfig/network-scripts/ifcfg-ens33'

重启生效
或者
[root@localhost ~]# source /etc/bashrc


```

查看目录个数

```shell
[root@localhost ~]# ls -l /etc/ | grep "^d" | wc -l
```
